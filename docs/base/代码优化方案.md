# 预约系统前端代码优化方案

## 1. 概述
本优化方案基于对预约系统前端代码的全面分析，并结合参考文档《READ.md》、《数据模型设计.md》、《前后端接口设计.md》、《前端架构设计.md》的要求，提出系统性的代码优化建议。本方案旨在解决现有代码与设计规范之间的不一致问题，提升代码质量、可维护性和性能。

### 3.4 状态管理优化

**优化内容**：
- 重构UIContext，完善主题切换功能，分离语言状态
- 优化Redux Toolkit配置，添加UI业务状态管理
- 明确状态管理边界，UI状态与业务状态分离
- 实现localStorage持久化主题设置

**具体实现**：
```tsx
// UIContext.tsx优化后的实现
import React, { createContext, useContext, useState, useEffect, useCallback, ReactNode } from 'react';

// 主题类型
export type ThemeType = 'light' | 'dark' | 'system';

// 获取系统主题
const getSystemTheme = (): ThemeType => {
  if (typeof window !== 'undefined') {
    const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
    return prefersDark ? 'dark' : 'light';
  }
  return 'light';
};

// 应用主题到文档
const applyThemeToDocument = useCallback((theme: ThemeType) => {
  // 根据主题类型设置类名
  const actualTheme = theme === 'system' ? getSystemTheme() : theme;
  
  // 添加Tailwind标准的dark类名
  if (actualTheme === 'dark') {
    document.documentElement.classList.add('dark');
    // 应用深色主题OKLCH色彩
    document.documentElement.style.setProperty('--primary-color', 'oklch(65.2% 0.251 10)');
    document.documentElement.style.setProperty('--background-color', 'oklch(12.3% 0 0)');
  } else {
    document.documentElement.classList.remove('dark');
    // 应用浅色主题OKLCH色彩
    document.documentElement.style.setProperty('--primary-color', 'oklch(65.2% 0.251 10)');
    document.documentElement.style.setProperty('--background-color', 'oklch(99.9% 0 0)');
  }
  
  // 保存主题设置到localStorage
  try {
    localStorage.setItem('app-theme', theme);
  } catch (error) {
    console.warn('Failed to save theme to localStorage:', error);
  }
}, []);

interface UIContextType {
  isDarkMode: boolean;
  isSidebarOpen: boolean;
  notifications: Notification[];
  toggleTheme: () => void;
  setSidebarOpen: (open: boolean) => void;
  addNotification: (notification: Omit<Notification, 'id' | 'createdAt'>) => void;
  removeNotification: (id: string) => void;
}

const UIContext = createContext<UIContextType | undefined>(undefined);

export const UIProvider: React.FC<{ children: ReactNode }> = ({ children }) => {
  // 设置默认深色主题
  const [isDarkMode, setIsDarkMode] = useState<boolean>(() => {
    const savedTheme = localStorage.getItem('theme');
    if (savedTheme) {
      return savedTheme === 'dark';
    }
    // 默认使用深色主题
    return true;
  });
  
  const [isSidebarOpen, setIsSidebarOpen] = useState<boolean>(false);
  const [notifications, setNotifications] = useState<Notification[]>([]);

  // 应用主题
  useEffect(() => {
    if (isDarkMode) {
      document.documentElement.classList.add('dark');
    } else {
      document.documentElement.classList.remove('dark');
    }
    localStorage.setItem('theme', isDarkMode ? 'dark' : 'light');
  }, [isDarkMode]);

  const toggleTheme = () => {
    setIsDarkMode(prev => !prev);
  };

  const addNotification = (notification: Omit<Notification, 'id' | 'createdAt'>) => {
    const newNotification: Notification = {
      ...notification,
      id: Date.now().toString(),
      createdAt: new Date().toISOString(),
    };
    setNotifications(prev => [...prev, newNotification]);
  };

  const removeNotification = (id: string) => {
    setNotifications(prev => prev.filter(notification => notification.id !== id));
  };

  const value: UIContextType = {
    isDarkMode,
    isSidebarOpen,
    notifications,
    toggleTheme,
    setSidebarOpen,
    addNotification,
    removeNotification,
  };

  return <UIContext.Provider value={value}>{children}</UIContext.Provider>;
};

// 统一使用useUI钩子，移除useUISafe钩子
export function useUI() {
  const context = useContext(UIContext);
  if (context === undefined) {
    throw new Error('useUI must be used within a UIProvider');
  }
  return context;
}
```

```tsx
// bookingSlice.ts优化示例
import { createSlice, createAsyncThunk, PayloadAction } from '@reduxjs/toolkit';
import { bookingService } from '../../services/bookingService';
import { Booking, BookingStatus } from '../../types';

interface BookingState {
  bookings: Booking[];
  loading: boolean;
  error: string | null;
}

const initialState: BookingState = {
  bookings: [],
  loading: false,
  error: null,
};

// 优化异步操作处理
export const fetchBookings = createAsyncThunk(
  'booking/fetchBookings',
  async (_, { rejectWithValue }) => {
    try {
      const response = await bookingService.getBookings();
      return response.data;
    } catch (error) {
      return rejectWithValue(error instanceof Error ? error.message : 'Failed to fetch bookings');
    }
  }
);

export const cancelBooking = createAsyncThunk(
  'booking/cancelBooking',
  async (bookingId: string, { rejectWithValue }) => {
    try {
      await bookingService.cancelBooking(bookingId);
      return bookingId;
    } catch (error) {
      return rejectWithValue(error instanceof Error ? error.message : 'Failed to cancel booking');
    }
  }
);

const bookingSlice = createSlice({
  name: 'booking',
  initialState,
  reducers: {},
  extraReducers: (builder) => {
    builder
      // 处理fetchBookings
      .addCase(fetchBookings.pending, (state) => {
        state.loading = true;
        state.error = null;
      })
      .addCase(fetchBookings.fulfilled, (state, action: PayloadAction<Booking[]>) => {
        state.loading = false;
        state.bookings = action.payload;
      })
      .addCase(fetchBookings.rejected, (state, action) => {
        state.loading = false;
        state.error = action.payload as string || 'Failed to fetch bookings';
      })
      // 处理cancelBooking
      .addCase(cancelBooking.pending, (state) => {
        state.loading = true;
        state.error = null;
      })
      .addCase(cancelBooking.fulfilled, (state, action: PayloadAction<string>) => {
        state.loading = false;
        state.bookings = state.bookings.map(booking => 
          booking.id === action.payload 
            ? { ...booking, status: BookingStatus.CANCELLED }
            : booking
        );
      })
      .addCase(cancelBooking.rejected, (state, action) => {
        state.loading = false;
        state.error = action.payload as string || 'Failed to cancel booking';
      });
  },
});

export default bookingSlice.reducer;
```

```typescript
import axios from 'axios';
import { ApiResponse, PaginatedResponse } from '../types';

// 创建axios实例
const api = axios.create({
  baseURL: process.env.NEXT_PUBLIC_API_BASE_URL || 'http://localhost:3001/api',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 请求拦截器
api.interceptors.request.use(
  (config) => {
    // 从localStorage获取token
    const token = localStorage.getItem('auth_token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器 - 统一处理响应格式
api.interceptors.response.use(
  (response) => {
    // 假设后端已经返回了符合ApiResponse格式的数据
    // 如果不是，这里可以进行转换
    return response.data;
  },
  (error) => {
    // 统一错误处理
    let errorMessage = 'Network error';
    
    if (error.response) {
      // 服务器返回错误
      if (error.response.data && error.response.data.message) {
        errorMessage = error.response.data.message;
      } else {
        errorMessage = `Error ${error.response.status}: ${error.response.statusText}`;
      }
      
      // 处理401未授权错误
      if (error.response.status === 401) {
        // 清除token并跳转到登录页
        localStorage.removeItem('auth_token');
        window.location.href = '/login';
      }
    } else if (error.request) {
      // 请求已发送但未收到响应
      errorMessage = 'No response from server';
    }
    
    // 创建一个标准化的错误对象
    return Promise.reject({
      success: false,
      code: error.response?.status || 500,
      message: errorMessage,
      data: null,
      timestamp: new Date().toISOString(),
      requestId: error.config?.headers['X-Request-ID'] || '',
    });
  }
);

export default api;
```

```typescript
import api from './api';
import { API_ENDPOINTS } from './endpoints';
import { VerifyLoginRequest, User, ApiResponse } from '../types';

// 发送验证码参数类型
export interface SendCodeRequest {
  phone: string;
  type: 'login' | 'register' | 'reset';
}

// 注册请求参数类型
export interface RegisterRequest {
  phone: string;
  code: string;
  name: string;
  password: string;
}

export const authService = {
  // 发送验证码
  async sendCode(data: SendCodeRequest): Promise<ApiResponse<{ success: boolean }>> {
    const response = await api.post(API_ENDPOINTS.AUTH.SEND_CODE, data);
    return response;
  },

  // 验证验证码并登录
  async verifyCode(data: VerifyLoginRequest): Promise<ApiResponse<{ token: string; user: User }>> {
    const response = await api.post(API_ENDPOINTS.AUTH.VERIFY_CODE, data);
    // 保存token到localStorage
    if (response.success && response.data.token) {
      localStorage.setItem('auth_token', response.data.token);
    }
    return response;
  },

  // 用户注册 (需要临时token)
  async register(data: RegisterRequest, tempToken: string): Promise<ApiResponse<{ token: string; user: User }>> {
    const response = await api.post(API_ENDPOINTS.AUTH.REGISTER, data, {
      headers: {
        'X-Temp-Token': tempToken,
      },
    });
    // 保存token到localStorage
    if (response.success && response.data.token) {
      localStorage.setItem('auth_token', response.data.token);
    }
    return response;
  },

  // 获取当前用户信息
  async getCurrentUser(): Promise<ApiResponse<User>> {
    const response = await api.get(API_ENDPOINTS.AUTH.ME);
    return response;
  },

  // 刷新token
  async refreshToken(): Promise<ApiResponse<{ token: string }>> {
    const response = await api.post(API_ENDPOINTS.AUTH.REFRESH);
    // 保存新token到localStorage
    if (response.success && response.data.token) {
      localStorage.setItem('auth_token', response.data.token);
    }
    return response;
  },

  // 登出
  async logout(): Promise<ApiResponse<{ success: boolean }>> {
    const response = await api.post(API_ENDPOINTS.AUTH.LOGOUT);
    // 清除token
    if (response.success) {
      localStorage.removeItem('auth_token');
    }
    return response;
  },
};
```
